-- 1. CREACIÓN DE LA BASE DE DATOS
CREATE DATABASE Inventario_Equipos;
GO
USE Inventario_Equipos;
GO

-- ========================================================
-- TABLAS MAESTRAS (CATÁLOGOS)
-- ========================================================

-- Tabla 1: Dependencias u Oficinas
CREATE TABLE T_DEPENDENCIAS (
    DEP_id INT IDENTITY(1,1),
    DEP_nombre VARCHAR(100) NOT NULL,
    CONSTRAINT PK_DEPENDENCIAS PRIMARY KEY (DEP_id),
    CONSTRAINT UQ_DEP_nombre UNIQUE (DEP_nombre)
);

-- Tabla 2: Usuarios / Personal responsable
CREATE TABLE T_USUARIOS (
    USU_id INT IDENTITY(1,1),
    USU_nombre VARCHAR(150) NOT NULL,
    DEP_id INT NOT NULL, -- Clave Foránea a Dependencias
    CONSTRAINT PK_USUARIOS PRIMARY KEY (USU_id),
    CONSTRAINT FK_USUARIOS_DEPENDENCIAS FOREIGN KEY (DEP_id) REFERENCES T_DEPENDENCIAS(DEP_id)
);

-- Tabla 3: Tipos de Partes / Periféricos (Monitor, Teclado, Mouse, Parlante)
CREATE TABLE T_TIPO_PARTES (
    TIP_par_id INT IDENTITY(1,1),
    TIP_par_descripcion VARCHAR(50) NOT NULL,
    CONSTRAINT PK_TIPO_PARTES PRIMARY KEY (TIP_par_id),
    CONSTRAINT UQ_TIP_par_desc UNIQUE (TIP_par_descripcion)
);

-- Tabla 4: Catálogo Maestro de Softwares existentes
CREATE TABLE T_CATALOGO_SOFTWARE (
    SOF_id INT IDENTITY(1,1),
    SOF_nombre VARCHAR(100) NOT NULL,
    SOF_categoria VARCHAR(50) NOT NULL, -- 'Sistema Operativo', 'Ofimática', 'Antivirus', etc.
    CONSTRAINT PK_CATALOGO_SOFTWARE PRIMARY KEY (SOF_id)
);

-- ========================================================
-- TABLA PRINCIPAL DEL MODELO
-- ========================================================

-- Tabla 5: Equipos Tecnológicos (Ficha Base)
CREATE TABLE T_EQUIPOS (
    EQU_id INT IDENTITY(1,1),
    EQU_cod_tecnologia VARCHAR(50) NOT NULL, -- Código de barras / Activo fijo
    EQU_cod_almacen VARCHAR(50) NULL,
    EQU_fecha_inventario DATE NOT NULL CONSTRAINT DF_EQU_fecha DEFAULT GETDATE(),
    EQU_tipo_hardware VARCHAR(30) NOT NULL,  -- 'Escritorio', 'Laptop', 'All In One'
    EQU_marca VARCHAR(50) NOT NULL,
    EQU_modelo VARCHAR(50) NOT NULL,
    EQU_num_serie VARCHAR(50) NOT NULL,
    
    -- Componentes Internos
    EQU_proc_codigo VARCHAR(50) NULL,
    EQU_proc_velocidad VARCHAR(20) NULL,
    EQU_ram_capacidad VARCHAR(20) NULL,
    EQU_disco_capacidad VARCHAR(20) NULL,
    EQU_disco_marca VARCHAR(50) NULL,
    EQU_unidad_cd_dvd CHAR(2) NOT NULL,      -- 'SI', 'NO'
    EQU_gpu_modelo VARCHAR(100) NULL,
    EQU_gpu_memoria VARCHAR(20) NULL,
    EQU_regulador CHAR(2) NOT NULL,          -- 'SI', 'NO'
    
    -- Estado y Asignación
    EQU_estado VARCHAR(15) NOT NULL,         -- 'BUENO', 'REGULAR', 'MALO'
    EQU_observacion VARCHAR(MAX) NULL,
    USU_id INT NOT NULL,                     -- Clave Foránea a Usuarios
    
    CONSTRAINT PK_EQUIPOS PRIMARY KEY (EQU_id),
    CONSTRAINT UQ_EQU_cod_tec UNIQUE (EQU_cod_tecnologia),
    CONSTRAINT UQ_EQU_num_ser UNIQUE (EQU_num_serie),
    CONSTRAINT FK_EQUIPOS_USUARIOS FOREIGN KEY (USU_id) REFERENCES T_USUARIOS(USU_id),
    CONSTRAINT CHK_EQU_tipo CHECK (EQU_tipo_hardware IN ('Escritorio', 'Laptop', 'All In One')),
    CONSTRAINT CHK_EQU_unidad CHECK (EQU_unidad_cd_dvd IN ('SI', 'NO')),
    CONSTRAINT CHK_EQU_regulador CHECK (EQU_regulador IN ('SI', 'NO')),
    CONSTRAINT CHK_EQU_estado CHECK (EQU_estado IN ('BUENO', 'REGULAR', 'MALO'))
);

-- ========================================================
-- TABLAS DE DETALLES Y RELACIONES COMPUESTAS
-- ========================================================

-- Tabla 6: Detalle de las Partes físicas asociadas al Equipo
CREATE TABLE T_EQUIPO_PARTES (
    EQU_id INT NOT NULL,
    TIP_par_id INT NOT NULL,
    PAR_marca VARCHAR(50) NULL,
    PAR_modelo VARCHAR(50) NULL,
    PAR_num_serie VARCHAR(50) NULL,
    CONSTRAINT PK_EQUIPO_PARTES PRIMARY KEY (EQU_id, TIP_par_id),
    CONSTRAINT FK_PARTES_EQUIPOS FOREIGN KEY (EQU_id) REFERENCES T_EQUIPOS(EQU_id) ON DELETE CASCADE,
    CONSTRAINT FK_PARTES_TIPO FOREIGN KEY (TIP_par_id) REFERENCES T_TIPO_PARTES(TIP_par_id)
);

-- Tabla 7: Detalle de Software Instalado en cada Equipo
CREATE TABLE T_EQUIPO_SOFTWARE (
    EQU_id INT NOT NULL,
    SOF_id INT NOT NULL,
    EQ_SOF_licencia CHAR(2) NOT NULL, -- 'SI', 'NO'
    CONSTRAINT PK_EQUIPO_SOFTWARE PRIMARY KEY (EQU_id, SOF_id),
    CONSTRAINT FK_EQSOF_EQUIPOS FOREIGN KEY (EQU_id) REFERENCES T_EQUIPOS(EQU_id) ON DELETE CASCADE,
    CONSTRAINT FK_EQSOF_SOFTWARE FOREIGN KEY (SOF_id) REFERENCES T_CATALOGO_SOFTWARE(SOF_id),
    CONSTRAINT CHK_EQSOF_licencia CHECK (EQ_SOF_licencia IN ('SI', 'NO'))
);

-- Tabla 8: Credenciales de Acceso Remoto para Soporte
CREATE TABLE T_ACCESO_REMOTO (
    EQU_id INT NOT NULL,
    REM_anydesk_codigo VARCHAR(30) NULL,
    REM_anydesk_contrasena VARCHAR(50) NULL,
    CONSTRAINT PK_ACCESO_REMOTO PRIMARY KEY (EQU_id),
    CONSTRAINT FK_REMOTO_EQUIPOS FOREIGN KEY (EQU_id) REFERENCES T_EQUIPOS(EQU_id) ON DELETE CASCADE
);

-- Tabla 9: Historial de Cambios de Estado (Para ańadir complejidad y control de auditoría)
CREATE TABLE T_HISTORIAL_ESTADOS (
    HIS_id INT IDENTITY(1,1),
    EQU_id INT NOT NULL,
    HIS_estado_anterior VARCHAR(15) NOT NULL,
    HIS_estado_nuevo VARCHAR(15) NOT NULL,
    HIS_fecha_cambio DATETIME NOT NULL CONSTRAINT DF_HIS_fecha DEFAULT GETDATE(),
    CONSTRAINT PK_HISTORIAL_ESTADOS PRIMARY KEY (HIS_id),
    CONSTRAINT FK_HISTORIAL_EQUIPOS FOREIGN KEY (EQU_id) REFERENCES T_EQUIPOS(EQU_id) ON DELETE CASCADE
);
GO

    EQU_modelo AS [Modelo Equipo],
    EQU_estado AS [Condición Actual]

-- ============================================================================
-- PROYECTO 6: Sumar 4 GB adicionales a la memoria RAM mostrada 
-- ============================================================================
-- Nota: Como EQU_ram_capacidad es VARCHAR (ej. '16 GB'), extraemos el valor
-- numérico usando TRY_CAST para poder operarlo aritméticamente sin romper el script.
SELECT 
    EQU_cod_tecnologia,
    EQU_ram_capacidad AS [RAM Original],
    CONCAT(TRY_CAST(SUBSTRING(EQU_ram_capacidad, 1, CHARINDEX(' ', EQU_ram_capacidad) - 1) AS INT) + 4, ' GB') AS [RAM Proyectada +4GB]
FROM T_EQUIPOS;
GO

-- ============================================================================
-- PROYECTO 7: Calcular el doble de almacenamiento 
-- ============================================================================
-- Nota: Extraemos el número de '512 GB SSD' o '1 TB HDD' para duplicarlo de forma segura.
SELECT 
    EQU_cod_tecnologia,
    EQU_disco_capacidad AS [Almacenamiento Original],
    TRY_CAST(SUBSTRING(EQU_disco_capacidad, 1, CHARINDEX(' ', EQU_disco_capacidad) - 1) AS INT) * 2 AS [Doble Almacenamiento Numérico]
FROM T_EQUIPOS;
GO

-- ============================================================================
-- PROYECTO 8: Demostrar cómo evitar errores de división (División segura) 
-- ============================================================================
-- Se utiliza NULLIF para transformar un posible divisor 0 en NULL, evitando el crash clásico.
SELECT 
    EQU_cod_tecnologia,
    EQU_marca,
    -- Simulación: Dividir la velocidad procesador entre un número variable (evitando división por cero)
    TRY_CAST(SUBSTRING(EQU_proc_velocidad, 1, CHARINDEX(' ', EQU_proc_velocidad) - 1) AS DECIMAL(5,2)) / NULLIF(0, 0) AS [Operación Div Segura]
FROM T_EQUIPOS;
GO

GO

-- ============================================================================

    EQU_cod_tecnologia,
    EQU_marca + ' - Obs: ' + ISNULL(EQU_observacion, '') AS [Método Tradicional con ISNULL],

-- ============================================================================
-- PROYECTO 24: Uso de IN con valores explícitos (Mostrar equipos HP y Dell) [cite: 95]
-- ============================================================================
SELECT 
    EQU_id, EQU_cod_tecnologia, EQU_marca, EQU_modelo 
FROM T_EQUIPOS
WHERE EQU_marca IN ('HP', 'Dell');
GO

-- ============================================================================
-- PROYECTO 25: Uso de IN con subconsulta simple [cite: 96]
-- Mostrar usuarios pertenecientes a dependencias existentes en el catálogo principal [cite: 98, 101]
-- ============================================================================
SELECT 
    USU_id, 
    USU_nombre, 
    DEP_id 
FROM T_USUARIOS
WHERE DEP_id IN (SELECT DEP_id FROM T_DEPENDENCIAS);
GO

-- ============================================================================
-- PROYECTO 26: Extraer ańo (Mostrar el ańo de registro) [cite: 85]
-- ============================================================================
SELECT 
    EQU_id,
    EQU_cod_tecnologia,
    EQU_fecha_inventario AS [Fecha Completa],
    YEAR(EQU_fecha_inventario) AS [Ańo del Inventario]
FROM T_EQUIPOS;
GO

-- ============================================================================
-- PROYECTO 27: Extraer mes y día [cite: 86]
-- ============================================================================
SELECT 
    EQU_id,
    EQU_fecha_inventario,
    MONTH(EQU_fecha_inventario) AS [Número de Mes],
    DAY(EQU_fecha_inventario) AS [Día del Mes]
FROM T_EQUIPOS;
GO

-- ============================================================================
-- PROYECTO 28: Agregar días a una fecha (Calcular fecha futura) [cite: 89]
-- Simulación: Sumar 180 días para proyectar la fecha del próximo mantenimiento obligatorio.
-- ============================================================================
SELECT 
    EQU_id,
    EQU_fecha_inventario AS [Fecha Registro],
    DATEADD(day, 180, EQU_fecha_inventario) AS [Próximo Mantenimiento Preventivo (180 días)]
FROM T_EQUIPOS;
GO

-- ============================================================================
-- PROYECTO 29: Diferencia entre fechas (Calcular días transcurridos) [cite: 91]
-- Calcula el tiempo transcurrido desde el inventario del equipo hasta la fecha actual del sistema.
-- ============================================================================
SELECT 
    EQU_id,
    EQU_fecha_inventario,
    GETDATE() AS [Fecha Servidor Actual],
    DATEDIFF(day, EQU_fecha_inventario, GETDATE()) AS [Días de Antigüedad en Sistema]
FROM T_EQUIPOS;
GO

-- ============================================================================
-- PROYECTO 30: Convertir formato de fechas (Mostrar fechas en distintos formatos) [cite: 100]
-- Empleo de la función CONVERT con los estándares internacionales más comunes.
-- ============================================================================
SELECT 
    EQU_fecha_inventario AS [Formato Nativo],
    CONVERT(VARCHAR, EQU_fecha_inventario, 103) AS [Formato Británico/Peruano (DD/MM/AAAA)],
    CONVERT(VARCHAR, EQU_fecha_inventario, 101) AS [Formato US (MM/DD/AAAA)],
    CONVERT(VARCHAR, EQU_fecha_inventario, 111) AS [Formato Japón (AAAA/MM/DD)]
FROM T_EQUIPOS;
GO
